# Config reference

This is the reference for `forgeui.config.ts` / `forgeui.config.js`.

If you want a starting point, run:

```bash
forgeui init
```

## Full example (defaults)

This is what `forgeui init` generates today:

```ts
import type { ForgeUIConfig } from "@forgeui/cli";

const config: ForgeUIConfig = {
  tokensPath: "./tokens.json",
  outDir: "./forgeui",
  plugins: [],
  themes: {
    rootTheme: "Light",
    selectorByTheme: {
      Light: ":root",
      Dark: ['[data-theme="dark"]', ".dark"],
    },
    fallbacks: {
      Dark: ["Light"],
    },
  },
  filter: {
    include: ["core.*", "components.*"],
    exclude: [],
    sets: [],
    types: [],
  },
  css: {
    alsoEmitClassDark: false,
    dimensions: {
      unit: "preserve",
      remBasePx: 16,
      precision: 4,
    },
  },
  tailwind: {
    cssFile: "tokens.css",
    presetFile: "forgeui.preset.ts",
    presetFormat: "esm",
    presetUsage: "v4",
    darkThemeName: "Dark",
    map: {
      colors: {},
    },
  },
  format: {
    prettier: false,
  },
};

export default config;
```

---

## Top-level

### `tokensPath`

Path to your Tokens Studio JSON export.

- Type: `string`
- Default: `"./tokens.json"`

### `outDir`

Where ForgeUI writes generated outputs.

- Type: `string`
- Default: `"./forgeui"`

### `plugins`

Plugins to run around generation.

- Type: `Array<{ module: string; name?: string; enabled?: boolean; options?: Record<string, unknown> }>`
- Default: `[]`

Each entry:

- `module`: either an npm package (e.g. `"@acme/forgeui-plugin"`) or a builtin short name (e.g. `"banner"`).
- `name` (optional): label used in logs/errors.
- `enabled` (optional): set `false` to disable without deleting the config.
- `options` (optional): free-form plugin options.

Example:

```ts
plugins: [
  { module: "banner", options: { text: "Generated by ForgeUI" } },
  { module: "@acme/forgeui-plugin", enabled: true, options: { foo: "bar" } },
],
```

### `themes`

Theme wiring and fallbacks.

#### `themes.rootTheme`

The canonical theme name used as the base for Tailwind mappings.

- Type: `string`
- Default: `"Light"`

#### `themes.selectorByTheme`

CSS selectors used for each theme.

- Type: `Record<string, string | string[]>`
- Default:

```ts
{
  Light: ":root",
  Dark: ['[data-theme="dark"]', ".dark"],
}
```

Notes:

- For the root theme, `:root` is usually what you want.
- For dark mode you can support multiple selectors (e.g. an attribute + a class).

#### `themes.fallbacks`

Theme fallback chain used when a token is missing in a theme.

- Type: `Record<string, string[]>`
- Default:

```ts
{ Dark: ["Light"] }
```

## `filter`

Controls which tokens are included in outputs.

### `filter.include`

Glob patterns matched against the fully-qualified token name (e.g. `core.color.bg.default`).

- Type: `string[]`
- Default: `['core.*', 'components.*']`

### `filter.exclude`

Glob patterns to exclude (applied after `include`).

- Type: `string[]`
- Default: `[]`

### `filter.sets`

Token *set* names (from Tokens Studio) to include. Empty means "all enabled sets" for each theme.

- Type: `string[]`
- Default: `[]`

### `filter.types`

Token types to include. Empty means "all".

- Type: `Array<string>`
- Default: `[]`

CLI overrides:

- `--types color,dimension`
- `--sets core,components`
- `--include core.*,components.*`
- `--exclude components.experimental.*`

---

## `css`

### `css.alsoEmitClassDark`

Also emit a `.dark { ... }` block (in addition to whatever you set for `themes.selectorByTheme.Dark`).

- Type: `boolean`
- Default: `false`

### `css.dimensions`

Controls how `$type: "dimension"` tokens are normalized.

- `unit`: `"preserve" | "px" | "rem"`
- `remBasePx`: `number` (default `16`)
- `precision`: `number` (default `4`)

Examples:

- Preserve units (recommended if your token source already contains units):

```ts
css: { dimensions: { unit: "preserve" } }
```

- Convert everything to `rem`:

```ts
css: { dimensions: { unit: "rem", remBasePx: 16, precision: 4 } }
```

## `tailwind`

Controls the generated Tailwind preset.

### `tailwind.cssFile`

Filename for generated CSS vars.

- Type: `string`
- Default: `"tokens.css"`

### `tailwind.presetFile`

Filename for the Tailwind preset.

- Type: `string`
- Default: `"forgeui.preset.ts"`

### `tailwind.themeFile` (optional)

If set, ForgeUI emits a separate Tailwind theme fragment file and imports it from the preset.

This is useful when you want the theme object available independently from the preset wrapper.

### `tailwind.presetFormat`

- `"esm"` (default)
- `"cjs"`

### `tailwind.presetUsage`

Controls the usage comment at the top of the generated preset.

- `"v4"` (default)
- `"v3"`

### `tailwind.darkThemeName`

Theme name used to drive Tailwind dark-mode values.

- Type: `string`
- Default: `"Dark"`

### `tailwind.map.colors`

Optional namespace remapping for colors.

This is a *key-path* remap inside `color.*`. It’s a longest-prefix match.

Example: map `bg.*` → `surface.*`:

```ts
tailwind: {
  map: {
    colors: {
      bg: "surface",
    },
  },
}
```

---

## `format`

### `format.prettier`

If `true`, generated TS outputs are prettier-formatted.

- Type: `boolean`
- Default: `false`
