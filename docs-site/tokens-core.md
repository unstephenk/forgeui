# Core tokens

This page renders `tokens.index.json` (generated by `forgeui docs`) filtered to the `core.*` namespace.

<div id="app">
  <p>Loading…</p>
</div>

<script type="module">
  const el = document.querySelector('#app');
  const DEFAULT_NS = 'core';

  function esc(s) {
    return String(s)
      .replaceAll('&', '&amp;')
      .replaceAll('<', '&lt;')
      .replaceAll('>', '&gt;')
      .replaceAll('"', '&quot;')
      .replaceAll("'", '&#039;');
  }

  function namespaceOf(token) {
    const s = String(token ?? '');
    const first = s.split('.')[0] || 'other';
    return first;
  }

  function tokenId(token) {
    return String(token ?? '')
      .trim()
      .toLowerCase()
      .replaceAll(/[^a-z0-9]+/g, '-')
      .replaceAll(/(^-|-$)/g, '');
  }

  async function copy(text) {
    try {
      await navigator.clipboard.writeText(String(text));
      return true;
    } catch {
      return false;
    }
  }

  function render({ data, query, th }) {
    const tokens = (data.tokens ?? []).filter(t => namespaceOf(t.token) === DEFAULT_NS);

    const themes = new Set();
    for (const t of tokens) for (const k of Object.keys(t.themes ?? {})) themes.add(k);
    const themeList = Array.from(themes);

    const selectedThemes = (th ? String(th).split(',').map(s => s.trim()).filter(Boolean) : themeList)
      .filter(x => themeList.includes(x));

    const q = (query ?? '').trim().toLowerCase();

    const filtered = tokens.filter(t => {
      if (!q) return true;
      const hay = [t.token, t.type, t.cssVar, ...Object.values(t.themes ?? {})].join(' ').toLowerCase();
      return hay.includes(q);
    });

    const head = selectedThemes.map(th => `<th>${esc(th)}</th>`).join('');

    const rows = filtered
      .map(t => {
        const cols = selectedThemes
          .map(th => {
            const v = (t.themes ?? {})[th] ?? '';
            return `<td>
              <div class="tok-cell">
                <code>${esc(v)}</code>
                <button class="tok-copy" data-copy="${esc(v)}" title="Copy value">Copy</button>
              </div>
            </td>`;
          })
          .join('');

        const id = tokenId(t.token);
        const parts = String(t.token ?? '').split('.');
        const nsBadge = parts[0] || 'other';
        const crumbs = parts.slice(1).join(' / ');

        return `<tr id="${esc(id)}">
          <td>
            <div class="tok-cell">
              <a class="tok-permalink" href="#${esc(id)}" title="Permalink">#</a>
              <span class="tok-badge">${esc(nsBadge)}</span>
              <div class="tok-token">
                <code>${esc(t.token)}</code>
                ${crumbs ? `<div class=\"tok-path\"><small>${esc(crumbs)}</small></div>` : ''}
              </div>
              <button class="tok-copy" data-copy="${esc(t.token)}" title="Copy token">Copy</button>
            </div>
          </td>
          <td><code>${esc(t.type)}</code></td>
          <td>
            <div class="tok-cell">
              <code>${esc(t.cssVar)}</code>
              <button class="tok-copy" data-copy="${esc(t.cssVar)}" title="Copy CSS var">Copy</button>
            </div>
          </td>
          ${cols}
        </tr>`;
      })
      .join('');

    el.innerHTML = `
      <style>
        .tok-controls { display: flex; gap: 12px; flex-wrap: wrap; align-items: center; margin: 12px 0 18px; }
        .tok-controls input { padding: 6px 8px; border: 1px solid var(--vp-c-divider); border-radius: 8px; background: var(--vp-c-bg); }
        .tok-themes { display: flex; gap: 10px; flex-wrap: wrap; align-items: center; margin-top: 4px; }
        .tok-theme { font-size: 12px; opacity: 0.9; user-select: none; }
        .tok-theme input { margin-right: 6px; }
        .tok-meta { opacity: 0.8; }
        .tok-cell { display: flex; gap: 8px; align-items: center; }
        .tok-token { display: flex; flex-direction: column; gap: 2px; }
        .tok-path { opacity: 0.75; }
        .tok-badge { font-size: 11px; padding: 2px 8px; border-radius: 999px; border: 1px solid var(--vp-c-divider); opacity: 0.9; }
        .tok-permalink { font-size: 12px; opacity: 0.5; text-decoration: none; padding: 2px 6px; border-radius: 8px; border: 1px solid transparent; }
        .tok-permalink:hover { opacity: 1; border-color: var(--vp-c-divider); }
        tr:target { outline: 2px solid var(--vp-c-brand-1); outline-offset: 4px; border-radius: 10px; }
        .tok-copy { font-size: 12px; padding: 2px 8px; border-radius: 999px; border: 1px solid var(--vp-c-divider); background: transparent; cursor: pointer; }
        .tok-copy:hover { border-color: var(--vp-c-brand-1); color: var(--vp-c-brand-1); }
      </style>

      <p class="tok-meta"><small>Generated: ${esc(data.generatedAt)}</small></p>

      <div class="tok-controls">
        <label style="min-width: 260px; flex: 1;">
          <span class="tok-meta"><small>Search</small></span><br />
          <input id="tok-q" placeholder="color.brand… / --core-color… / #fff" value="${esc(query ?? '')}" />
        </label>

        <div>
          <span class="tok-meta"><small>Themes</small></span><br />
          <div id="tok-themes" class="tok-themes">
            ${themeList
              .map(name => {
                const checked = selectedThemes.includes(name) ? 'checked' : '';
                return `<label class="tok-theme"><input type="checkbox" data-theme="${esc(name)}" ${checked} /> ${esc(name)}</label>`;
              })
              .join('')}
          </div>
        </div>

        <span class="tok-meta"><small>Showing ${filtered.length} / ${tokens.length}</small></span>
      </div>

      <table>
        <thead>
          <tr>
            <th>Token</th>
            <th>Type</th>
            <th>CSS Var</th>
            ${head}
          </tr>
        </thead>
        <tbody>
          ${rows}
        </tbody>
      </table>
    `;

    el.querySelector('#tok-q')?.addEventListener('input', (e) => {
      const v = e.target?.value ?? '';
      const u = new URL(location.href);
      if (v) u.searchParams.set('q', v); else u.searchParams.delete('q');
      history.replaceState(null, '', u);
      const uu = new URL(location.href);
      render({ data, query: v, th: uu.searchParams.get('th') ?? '' });
    });

    el.querySelectorAll('#tok-themes input[type="checkbox"]').forEach(cb => {
      cb.addEventListener('change', () => {
        const picked = Array.from(el.querySelectorAll('#tok-themes input[type="checkbox"]'))
          .filter(x => x.checked)
          .map(x => x.getAttribute('data-theme'))
          .filter(Boolean);

        const u = new URL(location.href);
        if (picked.length === themeList.length) u.searchParams.delete('th');
        else u.searchParams.set('th', picked.join(','));
        history.replaceState(null, '', u);

        const uu = new URL(location.href);
        render({ data, query: uu.searchParams.get('q') ?? '', th: uu.searchParams.get('th') ?? '' });
      });
    });

    el.querySelectorAll('button.tok-copy').forEach(btn => {
      btn.addEventListener('click', async () => {
        const text = btn.getAttribute('data-copy') ?? '';
        const ok = await copy(text);
        btn.textContent = ok ? 'Copied' : 'Copy failed';
        setTimeout(() => (btn.textContent = 'Copy'), 900);
      });
    });
  }

  async function main() {
    const res = await fetch('/tokens.index.json');
    const data = await res.json();

    const u = new URL(location.href);
    const query = u.searchParams.get('q') ?? '';

    const th = u.searchParams.get('th') ?? '';
    render({ data, query, th });
  }

  main().catch(err => {
    el.innerHTML = `<pre>${esc(err?.stack ?? err)}</pre>`;
  });
</script>
