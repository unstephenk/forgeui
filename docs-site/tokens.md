# Tokens

This page renders `tokens.index.json` (generated by `forgeui docs`).

<div id="app">
  <p>Loading…</p>
</div>

<script type="module">
  const el = document.querySelector('#app');

  function esc(s) {
    return String(s)
      .replaceAll('&', '&amp;')
      .replaceAll('<', '&lt;')
      .replaceAll('>', '&gt;')
      .replaceAll('"', '&quot;')
      .replaceAll("'", '&#039;');
  }

  function namespaceOf(token) {
    const s = String(token ?? '');
    const first = s.split('.')[0] || 'other';
    return first;
  }

  function tokenId(token) {
    return String(token ?? '')
      .trim()
      .toLowerCase()
      .replaceAll(/[^a-z0-9]+/g, '-')
      .replaceAll(/(^-|-$)/g, '');
  }

  async function copy(text) {
    try {
      await navigator.clipboard.writeText(String(text));
      return true;
    } catch {
      return false;
    }
  }

  function parseNsHint(rawQuery) {
    const raw = String(rawQuery ?? '');
    const m = raw.match(/\bns:([a-zA-Z0-9_-]+)\b/);
    if (!m) return { query: raw, nsHint: null };
    const nsHint = m[1];
    const query = raw.replace(m[0], '').trim();
    return { query, nsHint };
  }

  function render({ data, query, ns, th, sort, dir }) {
    const tokens = data.tokens ?? [];

    const themes = new Set();
    for (const t of tokens) for (const k of Object.keys(t.themes ?? {})) themes.add(k);
    const themeList = Array.from(themes);

    const selectedThemes = (th ? String(th).split(',').map(s => s.trim()).filter(Boolean) : themeList)
      .filter(x => themeList.includes(x));

    const parsed = parseNsHint(query);
    const nsEff = parsed.nsHint ?? ns;
    const q = (parsed.query ?? '').trim().toLowerCase();

    const filtered = tokens.filter(t => {
      if (nsEff && nsEff !== 'all' && namespaceOf(t.token) !== nsEff) return false;
      if (!q) return true;
      const hay = [t.token, t.type, t.cssVar, ...Object.values(t.themes ?? {})].join(' ').toLowerCase();
      return hay.includes(q);
    });

    const sortKey = sort || 'token';
    const sortDir = dir === 'desc' ? 'desc' : 'asc';

    const groups = new Map();
    for (const t of filtered) {
      const n = namespaceOf(t.token);
      if (!groups.has(n)) groups.set(n, []);
      groups.get(n).push(t);
    }

    const nsList = Array.from(new Set(tokens.map(t => namespaceOf(t.token)))).sort();

    const head = selectedThemes.map(th => `<th>${esc(th)}</th>`).join('');

    const groupHtml = Array.from(groups.entries())
      .sort(([a], [b]) => a.localeCompare(b))
      .map(([group, items]) => {
        const sortedItems = [...items].sort((a, b) => {
          const av = String(a?.[sortKey] ?? '');
          const bv = String(b?.[sortKey] ?? '');
          const cmp = av.localeCompare(bv);
          return sortDir === 'desc' ? -cmp : cmp;
        });

        const rows = sortedItems
          .map(t => {
            const cols = selectedThemes
              .map(th => {
                const v = (t.themes ?? {})[th] ?? '';
                return `<td>
                  <div class="tok-cell">
                    <code>${esc(v)}</code>
                    <button class="tok-copy" data-copy="${esc(v)}" title="Copy value">Copy</button>
                  </div>
                </td>`;
              })
              .join('');

            const id = tokenId(t.token);
            const parts = String(t.token ?? '').split('.');
            const nsBadge = parts[0] || 'other';
            const crumbs = parts.slice(1).join(' / ');

            return `<tr id="${esc(id)}">
              <td>
                <div class="tok-cell">
                  <a class="tok-permalink" href="#${esc(id)}" title="Permalink">#</a>
                  <span class="tok-badge">${esc(nsBadge)}</span>
                  <div class="tok-token">
                    <a href="token?token=${encodeURIComponent(String(t.token))}" title="Open token detail">
                      <code>${esc(t.token)}</code>
                    </a>
                    ${crumbs ? `<div class=\"tok-path\"><small>${esc(crumbs)}</small></div>` : ''}
                  </div>
                  <button class="tok-copy" data-copy="${esc(t.token)}" title="Copy token">Copy</button>
                </div>
              </td>
              <td><span class="tok-type tok-type-${esc(t.type)}">${esc(t.type)}</span></td>
              <td>
                <div class="tok-cell">
                  <code>${esc(t.cssVar)}</code>
                  <button class="tok-copy" data-copy="${esc(t.cssVar)}" title="Copy CSS var">Copy</button>
                </div>
              </td>
              ${cols}
            </tr>`;
          })
          .join('');

        return `
          <section class="tok-group">
            <h2>${esc(group)} <small>(${items.length})</small></h2>
            <table>
              <thead>
                <tr>
                  <th>Token</th>
                  <th>Type</th>
                  <th>CSS Var</th>
                  ${head}
                </tr>
              </thead>
              <tbody>
                ${rows}
              </tbody>
            </table>
          </section>
        `;
      })
      .join('');

    el.innerHTML = `
      <style>
        .tok-controls { display: flex; gap: 12px; flex-wrap: wrap; align-items: center; margin: 12px 0 18px; }
        .tok-controls input, .tok-controls select { padding: 6px 8px; border: 1px solid var(--vp-c-divider); border-radius: 8px; background: var(--vp-c-bg); }
        .tok-dir { padding: 6px 10px; border: 1px solid var(--vp-c-divider); border-radius: 8px; background: transparent; cursor: pointer; }
        .tok-dir:hover { border-color: var(--vp-c-brand-1); color: var(--vp-c-brand-1); }
        .tok-ns-chips { display: flex; gap: 6px; flex-wrap: wrap; align-items: center; margin-top: 6px; }
        .tok-chip { font-size: 12px; padding: 2px 10px; border-radius: 999px; border: 1px solid var(--vp-c-divider); background: transparent; cursor: pointer; opacity: 0.95; }
        .tok-chip[aria-pressed="true"] { border-color: var(--vp-c-brand-1); color: var(--vp-c-brand-1); }
        .tok-chip:hover { border-color: var(--vp-c-brand-1); }
        .tok-themes { display: flex; gap: 10px; flex-wrap: wrap; align-items: center; margin-top: 4px; }
        .tok-theme { font-size: 12px; opacity: 0.9; user-select: none; }
        .tok-theme input { margin-right: 6px; }
        .tok-meta { opacity: 0.8; }
        .tok-group h2 { margin-top: 22px; }
        .tok-cell { display: flex; gap: 8px; align-items: center; }
        .tok-token { display: flex; flex-direction: column; gap: 2px; }
        .tok-path { opacity: 0.75; }
        .tok-badge { font-size: 11px; padding: 2px 8px; border-radius: 999px; border: 1px solid var(--vp-c-divider); opacity: 0.9; }
        .tok-type { font-size: 11px; padding: 2px 8px; border-radius: 999px; border: 1px solid var(--vp-c-divider); opacity: 0.9; }
        .tok-type-color { border-color: color-mix(in oklab, var(--vp-c-brand-1), transparent 70%); }
        .tok-type-dimension { border-color: color-mix(in oklab, var(--vp-c-brand-2), transparent 70%); }
        .tok-type-shadow { border-color: color-mix(in oklab, var(--vp-c-brand-3), transparent 70%); }
        .tok-permalink { font-size: 12px; opacity: 0.5; text-decoration: none; padding: 2px 6px; border-radius: 8px; border: 1px solid transparent; }
        .tok-permalink:hover { opacity: 1; border-color: var(--vp-c-divider); }
        tr:target { outline: 2px solid var(--vp-c-brand-1); outline-offset: 4px; border-radius: 10px; }
        .tok-copy { font-size: 12px; padding: 2px 8px; border-radius: 999px; border: 1px solid var(--vp-c-divider); background: transparent; cursor: pointer; }
        .tok-copy:hover { border-color: var(--vp-c-brand-1); color: var(--vp-c-brand-1); }
      </style>

      <p class="tok-meta"><small>Generated: ${esc(data.generatedAt)}</small></p>

      <div class="tok-controls">
        <label>
          <span class="tok-meta"><small>Namespace</small></span><br />
          <select id="tok-ns">
            <option value="all">All (${tokens.length})</option>
            ${nsList.map(n => `<option value="${esc(n)}">${esc(n)}</option>`).join('')}
          </select>
          <div class="tok-ns-chips" aria-label="Namespace filters">
            ${nsList.map(n => `<button class="tok-chip" type="button" data-ns="${esc(n)}" aria-pressed="${String(nsEff === n)}">${esc(n)}</button>`).join('')}
          </div>
        </label>

        <label style="min-width: 260px; flex: 1;">
          <span class="tok-meta"><small>Search</small></span><br />
          <input id="tok-q" placeholder="core.color.brand… / --core-color… / #fff" value="${esc(query ?? '')}" />
        </label>

        <div>
          <span class="tok-meta"><small>Themes</small></span><br />
          <div id="tok-themes" class="tok-themes">
            ${themeList
              .map(name => {
                const checked = selectedThemes.includes(name) ? 'checked' : '';
                return `<label class="tok-theme"><input type="checkbox" data-theme="${esc(name)}" ${checked} /> ${esc(name)}</label>`;
              })
              .join('')}
          </div>
        </div>

        <label>
          <span class="tok-meta"><small>Sort</small></span><br />
          <select id="tok-sort">
            <option value="token">Token</option>
            <option value="type">Type</option>
            <option value="cssVar">CSS Var</option>
          </select>
        </label>

        <button id="tok-dir" class="tok-dir" title="Toggle sort direction">${sortDir === 'desc' ? '↓' : '↑'}</button>

        <span class="tok-meta"><small>Showing ${filtered.length} / ${tokens.length}</small></span>
      </div>

      ${groupHtml || '<p>No matches.</p>'}
    `;

    const sel = el.querySelector('#tok-ns');
    if (sel) sel.value = nsEff || 'all';

    el.querySelector('#tok-q')?.addEventListener('input', (e) => {
      const v = e.target?.value ?? '';
      const u = new URL(location.href);
      if (v) u.searchParams.set('q', v); else u.searchParams.delete('q');
      history.replaceState(null, '', u);
      const uu = new URL(location.href);
      render({ data, query: v, ns: uu.searchParams.get('ns') ?? 'all', th: uu.searchParams.get('th') ?? '', sort: uu.searchParams.get('sort') ?? 'token', dir: uu.searchParams.get('dir') ?? 'asc' });
    });

    el.querySelector('#tok-ns')?.addEventListener('change', (e) => {
      const v = e.target?.value ?? 'all';
      const u = new URL(location.href);
      if (v && v !== 'all') u.searchParams.set('ns', v); else u.searchParams.delete('ns');
      history.replaceState(null, '', u);
      const uu = new URL(location.href);
      render({ data, query: uu.searchParams.get('q') ?? '', ns: v, th: uu.searchParams.get('th') ?? '', sort: uu.searchParams.get('sort') ?? 'token', dir: uu.searchParams.get('dir') ?? 'asc' });
    });

    el.querySelectorAll('button.tok-chip').forEach(btn => {
      btn.addEventListener('click', () => {
        const v = btn.getAttribute('data-ns') || 'all';
        const u = new URL(location.href);
        if (v && v !== 'all') u.searchParams.set('ns', v); else u.searchParams.delete('ns');
        history.replaceState(null, '', u);
        const uu = new URL(location.href);
        render({ data, query: uu.searchParams.get('q') ?? '', ns: v, th: uu.searchParams.get('th') ?? '', sort: uu.searchParams.get('sort') ?? 'token', dir: uu.searchParams.get('dir') ?? 'asc' });
      });
    });

    const sortSel = el.querySelector('#tok-sort');
    if (sortSel) sortSel.value = sortKey;

    el.querySelector('#tok-sort')?.addEventListener('change', (e) => {
      const v = e.target?.value ?? 'token';
      const u = new URL(location.href);
      if (v && v !== 'token') u.searchParams.set('sort', v); else u.searchParams.delete('sort');
      history.replaceState(null, '', u);
      const uu = new URL(location.href);
      render({ data, query: uu.searchParams.get('q') ?? '', ns: uu.searchParams.get('ns') ?? 'all', th: uu.searchParams.get('th') ?? '', sort: uu.searchParams.get('sort') ?? 'token', dir: uu.searchParams.get('dir') ?? 'asc' });
    });

    el.querySelector('#tok-dir')?.addEventListener('click', () => {
      const u = new URL(location.href);
      const cur = u.searchParams.get('dir') === 'desc' ? 'desc' : 'asc';
      const next = cur === 'desc' ? 'asc' : 'desc';
      if (next === 'asc') u.searchParams.delete('dir'); else u.searchParams.set('dir', 'desc');
      history.replaceState(null, '', u);
      const uu = new URL(location.href);
      render({ data, query: uu.searchParams.get('q') ?? '', ns: uu.searchParams.get('ns') ?? 'all', th: uu.searchParams.get('th') ?? '', sort: uu.searchParams.get('sort') ?? 'token', dir: uu.searchParams.get('dir') ?? 'asc' });
    });

    el.querySelectorAll('#tok-themes input[type="checkbox"]').forEach(cb => {
      cb.addEventListener('change', () => {
        const picked = Array.from(el.querySelectorAll('#tok-themes input[type="checkbox"]'))
          .filter(x => x.checked)
          .map(x => x.getAttribute('data-theme'))
          .filter(Boolean);

        const u = new URL(location.href);
        if (picked.length === themeList.length) u.searchParams.delete('th');
        else u.searchParams.set('th', picked.join(','));
        history.replaceState(null, '', u);

        const uu = new URL(location.href);
        render({ data, query: uu.searchParams.get('q') ?? '', ns: uu.searchParams.get('ns') ?? 'all', th: uu.searchParams.get('th') ?? '', sort: uu.searchParams.get('sort') ?? 'token', dir: uu.searchParams.get('dir') ?? 'asc' });
      });
    });

    el.querySelectorAll('button.tok-copy').forEach(btn => {
      btn.addEventListener('click', async () => {
        const text = btn.getAttribute('data-copy') ?? '';
        const ok = await copy(text);
        btn.textContent = ok ? 'Copied' : 'Copy failed';
        setTimeout(() => (btn.textContent = 'Copy'), 900);
      });
    });
  }

  async function main() {
    const res = await fetch('tokens.index.json');
    const data = await res.json();

    const u = new URL(location.href);
    const query = u.searchParams.get('q') ?? '';
    const ns = u.searchParams.get('ns') ?? 'all';
    const th = u.searchParams.get('th') ?? '';
    const sort = u.searchParams.get('sort') ?? 'token';
    const dir = u.searchParams.get('dir') ?? 'asc';

    render({ data, query, ns, th, sort, dir });
  }

  main().catch(err => {
    el.innerHTML = `<pre>${esc(err?.stack ?? err)}</pre>`;
  });
</script>
