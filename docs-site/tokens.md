# Tokens

This page renders `tokens.index.json` (generated by `forgeui docs`).

<div id="app">
  <p>Loading…</p>
</div>

<script type="module">
  const el = document.querySelector('#app');

  function esc(s) {
    return String(s)
      .replaceAll('&', '&amp;')
      .replaceAll('<', '&lt;')
      .replaceAll('>', '&gt;');
  }

  async function main() {
    const res = await fetch('/tokens.index.json');
    const data = await res.json();
    const tokens = data.tokens ?? [];

    const themes = new Set();
    for (const t of tokens) for (const k of Object.keys(t.themes ?? {})) themes.add(k);
    const themeList = Array.from(themes);

    const rows = tokens
      .map(t => {
        const cols = themeList.map(th => `<td><code>${esc((t.themes ?? {})[th] ?? '')}</code></td>`).join('');
        return `<tr>
          <td><code>${esc(t.token)}</code></td>
          <td><code>${esc(t.type)}</code></td>
          <td><code>${esc(t.cssVar)}</code></td>
          ${cols}
        </tr>`;
      })
      .join('');

    const head = themeList.map(th => `<th>${esc(th)}</th>`).join('');

    el.innerHTML = `
      <p><small>Generated: ${esc(data.generatedAt)}</small></p>
      <table>
        <thead>
          <tr>
            <th>Token</th>
            <th>Type</th>
            <th>CSS Var</th>
            ${head}
          </tr>
        </thead>
        <tbody>
          ${rows}
        </tbody>
      </table>
    `;
  }

  main().catch(err => {
    el.innerHTML = `<pre>${esc(err?.stack ?? err)}</pre>`;
  });
</script>
